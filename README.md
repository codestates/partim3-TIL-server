# partim3-TIL-server

EXPRESS를 이용한 프로젝트 TIL 서버 코드

# 로컬 테스트 환경

npm install

npm install -g typeorm

.env 파일 생성후
환경변수 입력

TOKEN_SECRET =

PORT =

DB_CONNECTION = mysql
DB_HOST = localhost
DB_PORT = 3306
DB_USERNAME =
DB_PASSWORD =
DB_DATABASE = TIL

KAKAO_CLIENTID =
NAVER_CLIENTID =
GOOGLE_CLIENTID =

GITHUB_CLIENTID =
GITHUB_CLIENTSECRET =

npm run dev 로 테스트
